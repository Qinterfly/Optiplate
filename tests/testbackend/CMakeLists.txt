
find_package(Qt6 REQUIRED COMPONENTS Core Test)
find_package(Eigen3 CONFIG REQUIRED)

qt_add_executable(testbackend
    testbackend.h
    testbackend.cpp
)

    WIN32_EXECUTABLE TRUE
    MACOSX_BUNDLE TRUE
)

    Qt6::Core
    Qt6::Test
    Eigen3::Eigen
    backend
)

    ${CMAKE_BINARY_DIR}
)
